#Quick examples demonstrating how R uses factors

#Example #1
set1 <- data.frame(cat.var = factor(c("D", "A", "A", "B", "D",
    "C", "1", "0", "1", "b")))
set1

class(set1$cat.var)
levels(set1$cat.var)

#Example #2 - shows what happens if you have a numerical variable
set2 <- data.frame(num.var = c(0,1,0,0,1))
set2

class(set2$num.var)
levels(set2$num.var)
factor(set2$num.var)
levels(factor(set2$num.var))
